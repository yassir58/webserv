function parseConfig()
{
    while (i < configFile.size())
    {
        line(container) = split(configFile[i]);
        if (line.size() > 0 && line[0] == "http")
            parseHttpContext();
        else
            parseDirective(line, MAIN_CONTEXT);
        i++;
    }
}

function checkDirective(line(container), CONTEXT)
{
    if (line.size() < 2)
    {
        print "Directive must atleast have one value"
        exit
    }
    if (line[size() - 1] != ';')
    {
        print "Directive must end with ;"
        exit
    }
}
